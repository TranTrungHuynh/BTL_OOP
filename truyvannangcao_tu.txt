
đưa ra tên tổ chức ở Hanoi và được thành lập sau 1976.
PREFIX organization: <http://shadow.org/organization/>
SELECT ?object
WHERE {
	?subject organization:has_Label ?object.
	?subject organization:has_Headquarter "Hanoi, Vietnam".
	?subject organization:has_Date ?date.
 
	FILTER (year(?date) <= 1976)

}
